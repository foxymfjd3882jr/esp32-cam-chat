bool canAcceptClient() {
  if (connectedClients < MAX_CLIENTS) {
    connectedClients++;
    DBG_OUTPUT_PORT.println("Client connected. Total: " + String(connectedClients));
    return true;
  }
  return false;
}

void disconnectClient() {
  if (connectedClients > 0) {
    connectedClients--;
    DBG_OUTPUT_PORT.println("Client disconnected. Total: " + String(connectedClients));
  }
}

void returnOK() {
  server.send(200, "text/plain", "");
}

void returnFail(String msg) {
  server.send(500, "text/plain", msg + "\r\n");

// –§—É–Ω–∫—Ü–∏–∏ –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è–º–∏
bool userExists(String username) {
  for (const User& user : users) {
    if (user.username == username) {
      return true;
    }
  }
  return false;
}

User* getUser(String username) {
  for (User& user : users) {
    if (user.username == username) {
      return &user;
    }
  }
  return nullptr;
}

bool authenticateUser(String username, String password) {
  User* user = getUser(username);
  return user != nullptr && user->password == password;
}

// –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö –≤ –ø–∞–º—è—Ç—å
void saveUsersToPreferences() {
  preferences.begin("chat-users", false);
  preferences.putUInt("userCount", users.size());
  
  for (size_t i = 0; i < users.size(); i++) {
    String prefix = "user" + String(i);
    preferences.putString((prefix + "_username").c_str(), users[i].username);
    preferences.putString((prefix + "_password").c_str(), users[i].password);
    preferences.putString((prefix + "_displayName").c_str(), users[i].displayName);
  }
  preferences.end();
  DBG_OUTPUT_PORT.println("Users saved to preferences");
}

void loadUsersFromPreferences() {
  preferences.begin("chat-users", true);
  unsigned int userCount = preferences.getUInt("userCount", 0);
  
  users.clear();
  for (size_t i = 0; i < userCount; i++) {
    String prefix = "user" + String(i);
    User user;
    user.username = preferences.getString((prefix + "_username").c_str(), "");
    user.password = preferences.getString((prefix + "_password").c_str(), "");
    user.displayName = preferences.getString((prefix + "_displayName").c_str(), "");
    
    if (user.username.length() > 0) {
      users.push_back(user);
    }
  }
  preferences.end();
  DBG_OUTPUT_PORT.println("Loaded " + String(users.size()) + " users from preferences");
}

void saveMessagesToPreferences() {
  preferences.begin("chat-messages", false);
  preferences.putUInt("messageCount", messages.size());
  
  for (size_t i = 0; i < messages.size(); i++) {
    String prefix = "msg" + String(i);
    preferences.putString((prefix + "_from").c_str(), messages[i].fromUser);
    preferences.putString((prefix + "_to").c_str(), messages[i].toUser);
    preferences.putString((prefix + "_content").c_str(), messages[i].content);
    preferences.putULong((prefix + "_timestamp").c_str(), messages[i].timestamp);
  }
  preferences.end();
  DBG_OUTPUT_PORT.println("Messages saved to preferences");
}

void loadMessagesFromPreferences() {
  preferences.begin("chat-messages", true);
  unsigned int messageCount = preferences.getUInt("messageCount", 0);
  
  messages.clear();
  for (size_t i = 0; i < messageCount; i++) {

  DBG_OUTPUT_PORT.println("Loaded " + String(messages.size()) + " messages from preferences");
}

// API endpoints
void handleRegister() {
  if (!canAcceptClient()) {
    server.send(503, "application/json", "{\"error\":\"Server is full. Try again later.\"}");
    return;
  }

  if (server.method() != HTTP_POST) {
    server.send(405, "application/json", "{\"error\":\"Method not allowed\"}");
    disconnectClient();
    return;
  }

  String username = server.arg("username");
  String password = server.arg("password");
  String displayName = server.arg("display_name");

  if (username.length() == 0 || password.length() == 0) {
    server.send(400, "application/json", "{\"error\":\"Username and password required\"}");
    disconnectClient();
    return;
  }

  if (userExists(username)) {
    server.send(400, "application/json", "{\"error\":\"User already exists\"}");
    disconnectClient();
    return;
  }

  users.push_back({username, password, displayName});
  saveUsersToPreferences();
  server.send(200, "application/json", "{\"success\":\"User registered successfully\"}");
  disconnectClient();
}

void handleLogin() {
  if (!canAcceptClient()) {
    server.send(503, "application/json", "{\"error\":\"Server is full. Try again later.\"}");
    return;
  }

  if (server.method() != HTTP_POST) {
    server.send(405, "application/json", "{\"error\":\"Method not allowed\"}");
    disconnectClient();
    return;
  }

  String username = server.arg("username");
  String password = server.arg("password");

  if (authenticateUser(username, password)) {
    currentUser = username;
    server.send(200, "application/json", "{\"success\":\"Login successful\", \"user\":\"" + username + "\"}");
  } else {
    server.send(401, "application/json", "{\"error\":\"Invalid credentials\"}");
    disconnectClient();
  }
}

void handleLogout() {
  currentUser = "";
  server.send(200, "application/json", "{\"success\":\"Logged out\"}");
  disconnectClient();
}

void handleSearchUser() {
  if (!canAcceptClient()) {
    server.send(503, "application/json", "{\"error\":\"Server is full. Try again later.\"}");
    return;
  }

  if (currentUser.length() == 0) {
    server.send(401, "application/json", "{\"error\":\"Not authenticated\"}");
    disconnectClient();
    return;
  }

  String searchUsername = server.arg("username");
  if (searchUsername.length() == 0) {
    server.send(400, "application/json", "{\"error\":\"Username parameter required\"}");
    disconnectClient();
    return;
  }

  User* user = getUser(searchUsername);
  if (user != nullptr && user->username != currentUser) {
    String response = "{\"username\":\"" + user->username + "\", \"display_name\":\"" + user->displayName + "\"}";
    server.send(200, "application/json", response);
  } else {
    server.send(404, "application/json", "{\"error\":\"User not found\"}");
  }
  disconnectClient();
}

void handleSendMessage() {
  if (!canAcceptClient()) {
    server.send(503, "application/json", "{\"error\":\"Server is full. Try again later.\"}");
    return;
  }

  if (currentUser.length() == 0) {
    server.send(401, "application/json", "{\"error\":\"Not authenticated\"}");
    disconnectClient();
    return;
  }

  String toUser = server.arg("to_user");
  String content = server.arg("content");

  if (toUser.length() == 0 || content.length() == 0) {
    server.send(400, "application/json", "{\"error\":\"Recipient and message content required\"}");
    disconnectClient();
    return;
  }

  if (!userExists(toUser)) {
    server.send(404, "application/json", "{\"error\":\"Recipient user not found\"}");
    disconnectClient();
    return;
  }

  messages.push_back({currentUser, toUser, content, millis()});
  saveMessagesToPreferences();
  server.send(200, "application/json", "{\"success\":\"Message sent\"}");
  disconnectClient();
}

void handleGetMessages() {
  if (!canAcceptClient()) {
    server.send(503, "application/json", "{\"error\":\"Server is full. Try again later.\"}");
    return;
  }

  if (currentUser.length() == 0) {
    server.send(401, "application/json", "{\"error\":\"Not authenticated\"}");
    disconnectClient();
    return;
  }

  String withUser = server.arg("with_user");
  if (withUser.length() == 0) {
    server.send(400, "application/json", "{\"error\":\"with_user parameter required\"}");
    disconnectClient();
    return;
  }

  String response = "[";
  bool first = true;
  
  for (const Message& msg : messages) {
    if ((msg.fromUser == currentUser && msg.toUser == withUser) || 
        (msg.fromUser == withUser && msg.toUser == currentUser)) {
      if (!first) response += ",";
      first = false;
      
      response += "{\"from\":\"" + msg.fromUser + "\",";
      response += "\"to\":\"" + msg.toUser + "\",";
      response += "\"content\":\"" + msg.content + "\",";
      response += "\"timestamp\":" + String(msg.timestamp) + "}";
    }
  }
  
  response += "]";
  server.send(200, "application/json", response);
  disconnectClient();
}

void handleGetChats() {
  if (!canAcceptClient()) {
    server.send(503, "application/json", "{\"error\":\"Server is full. Try again later.\"}");
    return;
  }

  if (currentUser.length() == 0) {
    server.send(401, "application/json", "{\"error\":\"Not authenticated\"}");
    disconnectClient();
    return;
  }

  std::vector<String> chatPartners;
  
  for (const Message& msg : messages) {
    String partner = "";
    if (msg.fromUser == currentUser) {
      partner = msg.toUser;
    } else if (msg.toUser == currentUser) {
      partner = msg.fromUser;
    }
    
    if (partner.length() > 0 && partner != currentUser) {
      bool exists = false;
      for (const String& existing : chatPartners) {
        if (existing == partner) {
          exists = true;
          break;
        }
      }
      if (!exists) {
        chatPartners.push_back(partner);
      }
    }
  }

  String response = "[";
  for (size_t i = 0; i < chatPartners.size(); i++) {
    if (i > 0) response += ",";
    User* user = getUser(chatPartners[i]);
    String displayName = user != nullptr ? user->displayName : chatPartners[i];
    response += "{\"username\":\"" + chatPartners[i] + "\", \"display_name\":\"" + displayName + "\"}";
  }
  response += "]";
  
  server.send(200, "application/json", response);
  disconnectClient();
}

    handleGetChats();
  } else {
    server.send(404, "text/plain", "Not Found");
  }
}

void setup(void) {
  DBG_OUTPUT_PORT.begin(115200);
  DBG_OUTPUT_PORT.print("\n");

  loadUsersFromPreferences();
  loadMessagesFromPreferences();

  WiFi.softAP(ssid, password);
  DBG_OUTPUT_PORT.print("Access Point: ");
  DBG_OUTPUT_PORT.println(ssid);
  DBG_OUTPUT_PORT.print("IP address: ");
  DBG_OUTPUT_PORT.println(WiFi.softAPIP());

  if (MDNS.begin(host)) {
    DBG_OUTPUT_PORT.println("MDNS responder started");
  }

  server.on("/", handleRoot);
  server.on("/chat.html", handleChat);
  server.on("/api/register", handleRegister);
  server.on("/api/login", handleLogin);
  server.on("/api/logout", handleLogout);
  server.on("/api/search", handleSearchUser);
  server.on("/api/send_message", handleSendMessage);
  server.on("/api/messages", handleGetMessages);
  server.on("/api/chats", handleGetChats);
  
  server.onNotFound(handleNotFound);

  server.begin();
  DBG_OUTPUT_PORT.println("HTTP server started");

  if (users.size() == 0) {
    users.push_back({"user1", "pass1", "User One"});
    users.push_back({"user2", "pass2", "User Two"});
    saveUsersToPreferences();
  }
}

void loop(void) {
  server.handleClient();
  delay(2);
}
